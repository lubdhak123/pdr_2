function selectEmploymentType(selectedBtn) {
    const parent = selectedBtn.parentElement;
    const buttons = parent.querySelectorAll('button');
    
    const activeClasses = ['bg-white', 'shadow-sm', 'text-primary'];
    const inactiveClasses = ['text-on-surface-variant', 'hover:bg-slate-200/50'];

    buttons.forEach(btn => {
        // Reset all buttons to inactive
        btn.classList.remove(...activeClasses);
        btn.classList.add(...inactiveClasses);
    });

    // Set selected button to active
    selectedBtn.classList.add(...activeClasses);
    selectedBtn.classList.remove(...inactiveClasses);
}
